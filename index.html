import json import serial import time print("=== Sistema AgroNexus Inicializado ===") print("Conectando ao Arduino Mega... Certifique-se de fechar o Monitor Serial da IDE.") try: arduino = serial.Serial(PORTA_ARDUINO, BAUDRATE, timeout=1) time.sleep(2) # Aguarda o Arduino estabilizar print(f"Sucesso: AgroNexus conectado na porta {PORTA_ARDUINO}!") except Exception as e: print(f"Erro de conexão: Verifique a porta {PORTA_ARDUINO} ou se a IDE do Arduino está aberta.") print(f"Detalhes: {e}") exit() ARQUIVO_HTML = "index.html" try: while True: if arduino.in_waiting > 0: linha = arduino.readline().decode('utf-8').strip() # Verifica se recebemos o formato JSON do Arduino if linha.startswith('{') and linha.endswith('}'): try: dados = json.loads(linha) solo = dados.get("solo", 0) temp = dados.get("temp", 0.0) umid = dados.get("umid", 0.0) print(f"[AgroNexus] Solo: {solo}% | Temp: {temp}°C | Ar: {umid}%") # Layout HTML atualizado e elegante do AgroNexus html_content = f""" AgroNexus - Monitoramento Inteligente
🌱 AgroNexus
Painel Geral

Monitoramento em Tempo Real

Dados atualizados via conexão USB direta do Arduino Mega

🌱
Umidade do Solo
{solo}%
🌡️
Temperatura do Ar
{temp}°C
💧
Umidade do Ar
{umid}%
AgroNexus © 2026 - Conectado via Python Bridge
""" with open(ARQUIVO_HTML, "w", encoding="utf-8") as f: f.write(html_content) except Exception: pass time.sleep(0.5) except KeyboardInterrupt: print("\n[AgroNexus] Monitoramento encerrado pelo usuário.") arduino.close()
